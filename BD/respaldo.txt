create database biblioteca;
use biblioteca;

drop database biblioteca;
create table libro(
idlibro int auto_increment not null primary key,
autor varchar(70),
titulo varchar(70),
cantidad int,
estado tinyint
);

/*consultas*/
select * from libro;
select autor from libro;
select autor,titulo from libro;
